#include <stdio.h>
#include <stdlib.h>

int main() {
    FILE *file;
    char name[50];
    int marks;

    file = fopen("student_data.txt", "w");
    
    if (file == NULL) {
        perror("Error opening file");
        return EXIT_FAILURE;
    }

    for (int i = 0; i < 5; i++) {
        printf("Enter name of student %d: ", i + 1);
        
        if (scanf("%49s", name) != 1) {
            fprintf(stderr, "Invalid input. Exiting.\n");
            fclose(file);
            return EXIT_FAILURE;
        }

        printf("Enter marks of student %d: ", i + 1);
        
        if (scanf("%d", &marks) != 1) {
            fprintf(stderr, "Invalid input for marks. Exiting.\n");
            fclose(file);
            return EXIT_FAILURE;
        }

        fprintf(file, "Name: %s, Marks: %d\n", name, marks);
    }

    fclose(file);

    printf("Data successfully written to student_data.txt\n");

    return EXIT_SUCCESS;
}
